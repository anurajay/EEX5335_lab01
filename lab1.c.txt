#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <unistd.h>
#include <sys/wait.h>

int main(int argc, char *argv[]) {
    if (argc != 2) {
        printf("Usage: %s <non-negative integer>\n", argv[0]);
        return 1;
    }

    int n = atoi(argv[1]);
    if (n < 0) {
        printf("ERROR: invalid number. Please give a non-negative input.\n");
        return 1;
    }

    pid_t pid = fork();

    if (pid < 0) {
        perror("Fork failed");
        return 1;
    } 
    else if (pid == 0) {
        // Child process
        printf("Child process PID = %d\n", getpid());
        int a = 0, b = 1, c;

        for (int i = 0; i < n; i++) {
            printf("fib%d = %d\n", i, a);
            c = a + b;
            a = b;
            b = c;
        }
        printf("Child process terminated.\n");
    } 
    else {
        // Parent process
        printf("Parent process PID = %d\n", getpid());
        printf("Parent is waiting for child...\n");
        waitpid(pid, NULL, 0);
        printf("Parent process terminated.\n");
    }

    return 0;
}
